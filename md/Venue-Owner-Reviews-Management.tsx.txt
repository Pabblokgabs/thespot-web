}`}
>
<div className={`px-6 py-4 ${getRatingBgColor(review.rating)}`}>
<div className="flex items-center justify-between mb-2">
<div className="flex items-center">
<img
src={review.reviewerAvatar}
alt={review.reviewerName}
className="w-10 h-10 rounded-full mr-3 object-cover"
/>
<div>
<h3 className="font-medium text-gray-900">{review.reviewerName}</h3>
{selectedSpot === 'all' && (
<p className="text-sm text-gray-600 flex items-center">
<HomeOutlined className="mr-1" /> {review.spotName}
</p>
)}
</div>
</div>
<div className="text-right">
<div className={`text-lg font-bold ${getRatingColor(review.rating)}`}>
{review.rating}/5
</div>
<div className="text-xs text-gray-500">
{new Date(review.date).toLocaleDateString('en-US', {
year: 'numeric',
month: 'short',
day: 'numeric'
})}
</div>
</div>
</div>
<div className="mb-2">
<Rate disabled defaultValue={review.rating} />
</div>
{review.eventName && (
<Tag color="blue" className="mb-2">
<CalendarOutlined className="mr-1" /> {review.eventName}
</Tag>
)}
{review.replied && (
<Tag color="green" className="mb-2 ml-1">
<CheckCircleOutlined className="mr-1" /> Replied
</Tag>
)}
{review.flagged && (
<Tag color="red" className="mb-2 ml-1">
<FlagOutlined className="mr-1" /> Flagged
</Tag>
)}
</div>
<div className="px-6 py-4">
<div className="text-gray-700">
{review.text.length > 150 && !expandedReviews.includes(review.id) ? (
<>
{review.text.substring(0, 150)}...
<Button
type="link"
onClick={() => toggleExpandReview(review.id)}
className="p-0 h-auto !rounded-button whitespace-nowrap"
>
Read More
</Button>
</>
) : (
<>
{review.text}
{review.text.length > 150 && (
<Button
type="link"
onClick={() => toggleExpandReview(review.id)}
className="p-0 h-auto !rounded-button whitespace-nowrap"
>
Show Less
</Button>
)}
</>
)}
</div>
</div>
<div className="px-6 py-3 bg-gray-50 flex items-center justify-between">
<div>
<Tooltip title="Mark as helpful">
<Button
type="text"
icon={<LikeOutlined />}
className="mr-1 !rounded-button whitespace-nowrap"
>
{review.helpful}
</Button>
</Tooltip>
</div>
<div>
<Tooltip title="Reply to review">
<Button
type="text"
icon={<MessageOutlined />}
onClick={() => handleReply(review.id)}
className="mr-1 !rounded-button whitespace-nowrap"
disabled={review.replied}
>
Reply
</Button>
</Tooltip>
<Tooltip title="Flag review">
<Button
type="text"
icon={<FlagOutlined />}
onClick={() => handleReport(review.id)}
className="!rounded-button whitespace-nowrap"
disabled={review.flagged}
>
Flag
</Button>
</Tooltip>
</div>
</div>
</div>
))}
</div>
)}
{/* Pagination */}
{filteredReviews.length > reviewsPerPage && (
<div className="mt-8 flex justify-center">
<Pagination
current={currentPage}
total={filteredReviews.length}
pageSize={reviewsPerPage}
onChange={handlePageChange}
showSizeChanger={false}
/>
</div>
)}
</div>
{/* Recent Reviews Carousel */}
<div className="bg-white rounded-lg shadow-sm p-6 mb-8">
<h2 className="text-xl font-bold text-gray-900 mb-4">Recent High Ratings</h2>
<Swiper
modules={[SwiperPagination, Autoplay]}
spaceBetween={30}
slidesPerView={1}
breakpoints={{
640: {
slidesPerView: 2,
},
1024: {
slidesPerView: 3,
},
}}
pagination={{ clickable: true }}
autoplay={{ delay: 5000 }}
className="mySwiper"
>
{allReviews
.filter(review => review.rating >= 4)
.sort((a, b) => new Date(b.date).getTime() - new Date(a.date).getTime())
.slice(0, 6)
.map(review => (
<SwiperSlide key={review.id}>
<div className="bg-gray-50 rounded-lg p-4 h-48 flex flex-col">
<div className="flex items-center mb-2">
<img
src={review.reviewerAvatar}
alt={review.reviewerName}
className="w-8 h-8 rounded-full mr-2 object-cover"
/>
<div>
<h3 className="font-medium text-gray-900 text-sm">{review.reviewerName}</h3>
<p className="text-xs text-gray-500">{review.spotName}</p>
</div>
<div className="ml-auto">
<Rate disabled defaultValue={review.rating} className="text-sm" />
</div>
</div>
<p className="text-gray-700 text-sm line-clamp-4 flex-grow">
{review.text}
</p>
<div className="mt-2 text-xs text-gray-500">
{new Date(review.date).toLocaleDateString('en-US', {
year: 'numeric',
month: 'short',
day: 'numeric'
})}
</div>
</div>
</SwiperSlide>
))}
</Swiper>
</div>
</main>
{/* Footer */}
<footer className="bg-gray-800 text-white py-8">
<div className="max-w-7xl mx-auto px-4 sm:px-6 lg:px-8">
<div className="grid grid-cols-1 md:grid-cols-3 gap-8">
<div>
<h3 className="text-lg font-semibold mb-4">Event Discovery Platform</h3>
<p className="text-gray-400">
Helping venue owners connect with guests and manage their online presence effectively.
</p>
</div>
<div>
<h3 className="text-lg font-semibold mb-4">Quick Links</h3>
<ul className="space-y-2">
<li><a href="#" className="text-gray-400 hover:text-white cursor-pointer">Dashboard</a></li>
<li><a href="#" className="text-gray-400 hover:text-white cursor-pointer">Manage Venues</a></li>
<li><a href="#" className="text-gray-400 hover:text-white cursor-pointer">Analytics</a></li>
<li><a href="#" className="text-gray-400 hover:text-white cursor-pointer">Help Center</a></li>
</ul>
</div>
<div>
<h3 className="text-lg font-semibold mb-4">Contact Us</h3>
<p className="text-gray-400 mb-2">
Have questions about managing your reviews?
</p>
<Button type="primary" className="!rounded-button whitespace-nowrap">
Contact Support
</Button>
</div>
</div>
<div className="mt-8 pt-8 border-t border-gray-700 text-center text-gray-400">
<p>© 2025 Event Discovery Platform. All rights reserved.</p>
</div>
</div>
</footer>
{/* Reply Modal */}
<Modal
title="Reply to Review"
open={replyModalVisible}
onOk={submitReply}
onCancel={() => setReplyModalVisible(false)}
okText="Submit Reply"
cancelText="Cancel"
>
<p className="mb-4">
Your reply will be visible to the reviewer and all users viewing this review.
</p>
<TextArea
rows={4}
value={replyText}
onChange={e => setReplyText(e.target.value)}
placeholder="Write your response here..."
className="mb-2"
/>
</Modal>
{/* Report Modal */}
<Modal
title="Flag Review"
open={reportModalVisible}
onOk={submitReport}
onCancel={() => setReportModalVisible(false)}
okText="Submit Report"
cancelText="Cancel"
>
<p className="mb-4">
Please select a reason for flagging this review:
</p>
<Form layout="vertical">
<Form.Item>
<Radio.Group value={reportReason} onChange={e => setReportReason(e.target.value)}>
<Space direction="vertical">
<Radio value="inappropriate">Inappropriate content</Radio>
<Radio value="spam">Spam or misleading</Radio>
<Radio value="notrelevant">Not relevant to venue</Radio>
<Radio value="duplicate">Duplicate review</Radio>
<Radio value="other">Other reason</Radio>
</Space>
</Radio.Group>
</Form.Item>
{reportReason === 'other' && (
<Form.Item>
<TextArea rows={2} placeholder="Please specify the reason..." />
</Form.Item>
)}
</Form>
</Modal>
</div>
);
};
export default App
